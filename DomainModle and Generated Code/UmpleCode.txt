namespace ca.mcgill.ecse321.project321.model;

class User
{
  abstract;
  email;
  name;
  password;
}

class Customer
{
  isA User;
  phone;
  * Customer -> 1 Address;
}

class StoreOwner
{
  isA User;
}

class Employee
{
  isA User;
  enum EmployeeStatus {Sick, Inactive, Active}  
  EmployeeStatus status;
}

class Cart
{
  cartID;
  enum ShoppingType {Delivery, Pickup};
  ShoppingType type;
  * -<@> 1 Customer;
  * -> * CartItem;
  * -> 1 TimeSlot;
  1 <@>- 0..1 Order;
}

class Order
{
  orderID;
  Boolean completed;
  Date orderDate;
  Integer total;
  payment;
}

class Product
{
  enum PriceType {PER_UNIT, PER_KILOS}
  PriceType priceType;
  productID;
  isAvailableOnline;
  Integer price;
  Integer stock;
}

class CartItem
{
  Integer quantity;
  cartItemID;
  * -> 1 Product;
}

class TimeSlot
{
  timeslotID;
  Time startTime;
  Time endTime;
  Date date;
  Integer maxOrderPerSlot;
  * -> 1 Day;
}

class Day
{
    enum WeekDays {Monday, Tuesday, Wednesday, Thursday, Friday, Saturday, Sunday} 
  WeekDays day;
  Time storeStartHour;
  Time storeEndHour;
}

class Shift
{
  shiftID;
  Time startHour;
  Time endHour;
  Date date;
  * -> 1 Day;
  * -> 1 Employee;
}

class Address
{
  town;
  street;
  postalCode;
  Integer unit;
}

class TheGroceryStoreSystem
{
  1 <@>- 1 StoreOwner;
  1 <@>- * Employee;
  1 <@>- * Customer;
  1 <@>- * Product;
  1 <@>- * TimeSlot;
  1 <@>- * Shift;
  1 <@>- 7 Day;
  1 <@>- 1 Address;
  telephone;
  email;
  grocerystoreID;
}
