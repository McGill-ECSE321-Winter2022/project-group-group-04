lass User
{
  abstract;
  autounique userID;
  email;
  name;
  password;
}

class Customer
{
  isA User;
  phone;
  * Customer -> 1 Address;
}

class StoreOwner
{
  isA User;
}

class Employee
{
  isA User;
  enum EmployeeStatus {Sick, Inactive, Active}  
  EmployeeStatus status;
  * -> 1 Shift;
}

class Cart
{
  autounique cartID;
  ShoppingType type;
  1 -<@> 1 Customer;
  * -> * CartItem;
  * -> 1 TimeSlot;
  1 <@>- * Order;
}

class Order
{
  autounique orderID;
  Boolean completed;
  Date orderDate;
  Integer total;
  payment;
}

class Product
{
  autounique productID;
  isAvailableOnline;
  Integer price;
  Integer stock;
}

class CartItem
{
  Integer quantity;
  * -> 1 Product;
}

class TimeSlot
{
  Time startTime;
  Time endTime;
  Date date;
  Integer maxOrderPerSlot;
  enum ShoppingType {Delivery, Pickup}  
  ShoppingType shoppingType;
  * -> 1 Day;
}

{
    enum WeekDays {Monday, Tuesday, Wednesday, Thursday, Friday, Saturday, Sunday}

  Time startHour;
  Time endHour;
  Date date;
  * -> 1 Day;
}

class Address

  street;
  postalCode;
  Integer unit;
}
